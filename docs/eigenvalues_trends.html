Eigenvalues trend plots

Kushal K Dey

3/31/2018

  • Example
  • Plotting Hub eigenvalues
  • Plotting Toeplitz eigenvalues
  • Plotting banded precision eigenvalues

Here we plot the trends of the eigenvalues of the estimated correlation matrices using different approaches - CorShrink, glasso and corpcor - against the true correlation matrix.

library(ggplot2)
library(gridExtra)
hub_eigenvalues <- get(load("../output/eigenvalues_distribution/hub_eigenvalues_distribution.rda"))
banded_prec_eigenvalues <- get(load("../output/eigenvalues_distribution/banded_precision_eigenvalues_distribution.rda"))
toeplitz_eigenvalues <- get(load("../output/eigenvalues_distribution/toeplitz_eigenvalues_distribution.rda"))

Example

eigenvalue_trends <- hub_eigenvalues
num_samp <- c(10, 50, 100, 1000)
num <- 20
eigs.df <- data.frame ("x" = rep(1:num, 5),
                         "y" = eigenvalue_trends[[1]]$mean,
                         "color" = c(rep("empirical", num), 
                                     rep("corpcor",  num), rep("glasso", num),
                                     rep("corshrink", num), rep("true", num)),
                         "type" = c(rep("A", 4*num), rep("B", num)))

ggplot(eigs.df, aes(x=x, y=y, colour=color, linetype = color)) + geom_line(lty = 1, lwd = 0.7) +
    scale_linetype_manual(values = c(rep("solid", 4), rep("dashed", 1))) +
    scale_colour_manual(values=c("#000000", "blue", "green", "gold",
                                 "red", "#0072B2", "#CC79A7", "#F0E442")) + xlab("") + ylab("eigenvalues") +
    theme_bw()

Plotting Hub eigenvalues

eigenvalue_trends <- hub_eigenvalues
num_samp <- c(10, 50, 100, 1000)
num <- 20
gg <- list()
for(i in 1:4){
  eigs.df <- data.frame ("x" = rep(1:num, 5),
                         "y" = eigenvalue_trends[[i]]$mean,
                         "color" = c(rep("empirical", num), 
                                     rep("corpcor",  num), rep("glasso", num),
                                     rep("corshrink", num), rep("true", num)),
                         "type" = c(rep("A", 4*num), rep("B", num)))

  gg[[i]] <- ggplot(eigs.df, aes(x=x, y=y, colour=color, linetype = color)) + geom_line(lty = 1, lwd = 0.7) +
    scale_linetype_manual(values = c(rep("solid", 4), rep("dashed", 1))) +
    scale_colour_manual(values=c("#000000", "blue", "green", "gold",
                                 "red", "#0072B2", "#CC79A7", "#F0E442")) + xlab("") + ylab("eigenvalues") + ggtitle(paste0("n = ", num_samp[i], ", p = 100")) + 
    theme_bw() + theme(legend.position="none")
}

grid.arrange(gg[[1]], gg[[2]], gg[[3]],
             gg[[4]], nrow = 2, ncol = 2)
